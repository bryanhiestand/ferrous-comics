A friend pointed me at his CI:
name: Build and Push Docker Image

on:
  workflow_dispatch:
  push:
    branches:
      - master
  pull_request:
    types: [opened, synchronize, reopened, labeled]
  schedule:
    # Rebuild weekly to pick up base image security patches
    - cron: '0 0 */6 * *'

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: org/and-repo-go-here

jobs:
  # ── CI checks (fmt, clippy, test, doc, audit, machete) ──────────────

  rust-ci:
    runs-on: ubuntu-24.04-arm
    timeout-minutes: 15
    steps:
      - name: Checkout repository
        uses: actions/checkout@v6

      - name: Install Rust toolchain
        uses: dtolnay/rust-toolchain@stable
        with:
          components: rustfmt, clippy

      - name: Install cargo-binstall
        uses: cargo-bins/cargo-binstall@main

      - name: Install CI tools
        run: cargo binstall --no-confirm cargo-audit cargo-machete

      - name: Cache dependencies
        uses: Swatinem/rust-cache@v2

      - name: Check formatting
        run: cargo fmt --all -- --check

      - name: Clippy
        run: cargo clippy --workspace --tests -- -D warnings

      - name: Tests
        run: cargo test --workspace

      - name: Docs
        run: cargo doc --workspace --no-deps

      - name: Security audit
        run: cargo audit

      - name: Unused dependencies
        run: cargo machete

  # ── Build and push Docker image (runs after CI passes) ──────────────

  build-docker:
    needs: [rust-ci]
    if: github.event_name != 'pull_request' || contains(github.event.pull_request.labels.*.name, 'deploy')
    runs-on: ubuntu-24.04
    timeout-minutes: 20
    permissions:
      contents: read
      packages: write

    steps:
      - name: Checkout repository
        uses: actions/checkout@v6

      - name: Log in to GitHub Container Registry
        uses: docker/login-action@v4
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata for Docker
        id: meta
        uses: docker/metadata-action@v6
        with:
          images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}
          tags: |
            type=raw,value=latest

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v4

      - name: Build and push Docker image
        uses: docker/build-push-action@v7
        with:
          context: .
          platforms: linux/amd64
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: |
            type=gha,scope=${{ github.ref_name }}
            type=gha,scope=master
          cache-to: type=gha,mode=max,scope=${{ github.ref_name }}
